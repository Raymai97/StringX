// Prototype
STRX_CHAR *strxCat(const STRX_CHAR *str, ...);

// NOTE:
// Highly recommended to use strxCAT macro, not the function itself.

/* Do you know C# / Java ?
   string str = "str1 " + "str2 " + "str3 ";
  
   This is how you do with StringX:
   char *ret = strxCAT(str1, str2, str3);
   
   Yes. You don't have to tell how many arguments you're passing.
   This is the special feature of strxCAT. ;D */

// If you use strxCat...
char *s1 = strxCat("Hello", "World", "Foobar"); // WRONG
char *s2 = strxCat("Hello", "World", "Foobar", ""); // OK
// To avoid accident, use macro
char *s3 = strxCAT("Hello", "World", "Foobar"); // OK

// Let's see how the dream comes true
char *output, *name, *company, *hobby;
// name = GetUserName();
// company = GetUserCompany();
// hobby = GetUserHobby();

output = strxCAT(
	"Hello, ", name, "! How is your life as a worker of ", company, " today?\n",
	"While hardworking is good, please remember that you deserved to \n", 
	"do your favorite thing such as ", hobby, " too. Have a nice day! :) \n\n");

printf(output); // all out at once
