// Prototype
size_t strxIndexOf(const STRX_CHAR *str, const STRX_CHAR *key, STRX_FLAGS f);

// NOTE:
// If can't find the key, return value IS NOT -1, but the length of str.

// Example
STRX_FLAGS f = STRX_CASE_INSENSITIVE;
const char *str = "One does not simply foo the bar.";
size_t i;

// Find first 'o'
i = strxIndexOf(str, "o", f);
// Find 'o' after 10 chars
i = strxIndexOf(str + 10, "o", f);
// Find any 'a' or 'e' after i
i = strxIndexOf(str + i, "ae", f | STRX_KEY_AS_SINGLE_CHAR);

// It can be used like "StartsWith" function too.
char input[255];
// input = GetUserInput();
if (strxIndexOf(input, "why", f) == 0)
{
	// The input starts with "why"
}
else
{
	// Do something else
}
