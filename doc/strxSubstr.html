// Prototype
STRX_CHAR *strxSubstr(const STRX_CHAR *str, int offset, size_t count);

// Do you know this song?
const char *str = "Zankoku na Tenshi no you ni, Shounen yo Shinwa ni nare";

// Substring from the last 25 chars for 10 chars
printf("%s ", strxSubstr(str, -25, 10));

// Substring from index 11 for 6 chars
printf("%s ", strxSubstr(str, 11, 6));

// Substring the last 7 chars
printf("%s \n\n", strxSubstr(str, -7, 7));
