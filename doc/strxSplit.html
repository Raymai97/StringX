// Prototype
size_t strxSplit(const STRX_CHAR *str, const STRX_CHAR *delim, STRX_CHAR ***strs, STRX_FLAGS f);

// NOTE:
// Return value is number of elements of strs (string array).
// If str or delim is empty string, return value will be 0, and strs will NOT be modified.

STRX_FLAGS f = STRX_CASE_INSENSITIVE;
const char *str = "The art of troll is simple, first you must ..., then you must ...";
char **strArray;
size_t i, strCount;

// Split by " " (whitespace)
strCount = strxSplit(str, " ", &strArray, f);
printf("Split string ... Count: %i\n", strCount);
for (i = 0; i < strCount; i++)
{
	printf("[%3i] %s\n", i + 1, strArray[i]);
}

// Split by " ,."
strCount = strxSplit(str, " ,.", &strArray, f | STRX_KEY_AS_SINGLE_CHAR);
printf("Split string ... Count: %i\n", strCount);
for (i = 0; i < strCount; i++)
{
	printf("[%3i] %s\n", i + 1, strArray[i]);
}
